77bool updateCtrl(struct repeating_timer *t) {
78 int32_t tmp;
79 float FMmod;
80 static float yA = 0;
81 static float yB = 0;
82 static float yD = 0;
83
84 lfo1.update();
85 eg1.update();
86 eg2.update();
87
88 //Limits VCA output level to prevent saturation
89 as.set_Vca_cv(velocity * VelToVca * volume * eg1.veg_f / (128 * 128 * 128));
90
91 as.set_Filter_freq_cv(Filter_freq + (eg2.veg_a * Filter_env / 8) + ((key - 64) * Filter_key * 2)
92 + ((lfo1.vlfo * Lfo1ToFilter) / 4) + (velocity * VelToFilter / 8));
93 as.set_Filter_res_cv(Filter_res + ((lfo1.vlfo * Lfo1ToRes) / 4));
94
95 as.set_DcoA_pw_cv(pwmA + (lfo1.vlfo * Lfo1ToPwmA / 128));
96 as.set_DcoB_pw_cv(pwmB + (lfo1.vlfo * Lfo1ToPwmB / 128));
97
98 dso.setPw(PwDSO + (lfo1.vlfo * Lfo1ToPwmDso / 128));
99
100 tmp = bend + (lfo1.vlfo * Lfo1ToFreq / 128) + (eg2.veg_a * Eg2ToFreq / 128) + GlbDetune;
101 FMmod = pow(2, tmp / 1200.0);
102
103 //Glide
104 if (Glide) {
105 yA += (freqA - yA) / Glide;
106 yB += (freqB - yB) / Glide;
107 yD += (freqDSO - yD) / Glide;
108 }
109 else {
110 yA = freqA;
111 yB = freqB;
112 yD = freqDSO;
113 }
114
115 as.set_DcoA_freq(yA * FMmod, RAZPid);
116 as.set_DcoB_freq(yB * FMmod, RAZPid);
117 dso.setFrequency(yD * FMmod);
118
119 as.set_Pan_cv(Pan + (lfo1.vlfo * Lfo1ToPan / 128) + ((key - 64) * keyToPan / 128));
120
121 RAZPid = false;
122 return true;
123}
